Перезапуск сервера
	adb kill-server
	adb start-server

Запуск без snapshot (state прошлого запуска)
	emulator -avd <device-name> -no-snapshot-load -no-boot-anim

Удаляем снапшот
	del "%USERPROFILE%\.android\avd\<name_of_device>.avd\snapshots\default_boot\snapshot.pb"

Перегенерация adbkey 
	del %USERPROFILE%\.android\adbkey*
	adb kill-server && adb start-server
